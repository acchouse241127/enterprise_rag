```python
# backend/app/document_parser/models.py
from dataclasses import dataclass, field
from enum import Enum
from typing import Any

class ContentType(Enum):
    TEXT = "text"
    TABLE = "table"
    IMAGE = "image"
    EQUATION = "equation"
    AUDIO = "audio"
    VIDEO = "video"

@dataclass
class ParsedContent:
    """解析后的结构化内容单元"""
    content_type: ContentType
    text: str  # 主要文本内容
    metadata: dict[str, Any] = field(default_factory=dict)
    page_number: int | None = None
    position: dict[str, Any] | None = None  # {x, y, width, height}

    # metadata 常用字段:
    # - table_markdown: 表格的 Markdown 表示
    # - table_html: 表格的 HTML 表示
    # - image_path: 图片路径
    # - vlm_description: VLM 生成的图片描述
    # - latex: 公式的 LaTeX 表示
    # - ocr_text: OCR 识别的文本
```

#### 2.1.2 BaseDocumentParser 接口变更

```python
# backend/app/document_parser/base.py
from abc import ABC, abstractmethod
from pathlib import Path
from .models import ParsedContent

class BaseDocumentParser(ABC):
    """文档解析器抽象基类"""

    @abstractmethod
    def parse(self, file_path: Path) -> list[ParsedContent]:
        """解析文档，返回结构化内容列表"""
        raise NotImplementedError

    def parse_text(self, file_path: Path) -> str:
        """向后兼容：仅返回纯文本"""
        contents = self.parse(file_path)
        return "\n".join(c.text for c in contents if c.text)
```

#### 2.1.3 转换层

```python
# backend/app/document_parser/converter.py
from .models import ParsedContent, ContentType

def content_list_to_chunks(
    contents: list[ParsedContent],
    include_type_prefix: bool = True,
) -> list[str]:
    """将 ParsedContent 列表转换为带类型标记的文本 chunk"""
    chunks = []
    for c in contents:
        text = c.text
        if not text:
            continue

        prefix = ""
        if include_type_prefix:
            type_prefixes = {
                ContentType.TEXT: "",
                ContentType.TABLE: "[表格]\n",
                ContentType.IMAGE: "[图片]",
                ContentType.EQUATION: "[公式] ",
            }
            prefix = type_prefixes.get(c.content_type, "")

        # 处理特殊类型
        if c.content_type == ContentType.TABLE:
            # 优先使用 Markdown 格式
            table_md = c.metadata.get("table_markdown", "")
            if table_md:
                text = table_md
        elif c.content_type == ContentType.IMAGE:
            # 合并 VLM 描述和 OCR 文本
            vlm_desc = c.metadata.get("vlm_description", "")
            if vlm_desc:
                text = f"{vlm_desc}\n{text}"

        chunk = f"{prefix}{text}".strip()
        if chunk:
            chunks.append(chunk)

    return chunks

def get_content_metadata(contents: list[ParsedContent]) -> dict:
    """提取内容元数据用于向量存储"""
    return {
        "has_table": any(c.content_type == ContentType.TABLE for c in contents),
        "has_image": any(c.content_type == ContentType.IMAGE for c in contents),
        "has_equation": any(c.content_type == ContentType.EQUATION for c in contents),
        "page_count": max((c.page_number or 1 for c in contents), default=1),
    }
```

### 3.3 实现示例

```python
# backend/app/document_parser/docling_pdf_parser.py
from pathlib import Path
from docling.document_converter import DocumentConverter
from .base import BaseDocumentParser
from .models import ParsedContent, ContentType

class DoclingPdfParser(BaseDocumentParser):
    """基于 Docling 的 PDF 解析器"""

    def __init__(self):
        self.converter = DocumentConverter()

    def parse(self, file_path: Path) -> list[ParsedContent]:
        result = self.converter.convert(str(file_path))
        doc = result.document

        contents = []
        for item in doc.iterate_items():
            if item.label == "text":
                contents.append(ParsedContent(
                    content_type=ContentType.TEXT,
                    text=item.text,
                    page_number=item.page_no,
                ))
            elif item.label == "table":
                table_md = item.export_to_markdown()
                contents.append(ParsedContent(
                    content_type=ContentType.TABLE,
                    text=table_md,
                    metadata={"table_markdown": table_md},
                    page_number=item.page_no,
                ))
            elif item.label == "formula":
                contents.append(ParsedContent(
                    content_type=ContentType.EQUATION,
                    text=item.text,
                    metadata={"latex": item.text},
                    page_number=item.page_no,
                ))

        return contents
```

```python
# backend/app/llm/vlm_base.py
from abc import ABC, abstractmethod
from pathlib import Path

class BaseVLMClient(ABC):
    """视觉语言模型客户端基类"""

    @abstractmethod
    def describe_image(self, image_path: Path, prompt: str | None = None) -> str:
        """生成图片描述"""
        raise NotImplementedError

# backend/app/llm/vlm_openai.py
import base64
from openai import OpenAI
from .vlm_base import BaseVLMClient

class OpenAIVLMClient(BaseVLMClient):
    def __init__(self, api_key: str, model: str = "gpt-4o-mini"):
        self.client = OpenAI(api_key=api_key)
        self.model = model

    def describe_image(self, image_path: Path, prompt: str | None = None) -> str:
        with open(image_path, "rb") as f:
            image_data = base64.b64encode(f.read()).decode()

        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {
                    "role": "user",
                    "content": [
                        {"type": "text", "text": prompt or "请描述这张图片的内容，包括其中的关键信息。"},
                        {"type": "image_url", "image_url": {"url": f"data:image/jpeg;base64,{image_data}"}},
                    ],
                }
            ],
            max_tokens=500,
        )
        return response.choices[0].message.content or ""
```

```python
# backend/app/rag/query_analyzer.py
import re

class QueryAnalyzer:
    """分析查询意图，识别是否需要特定模态"""

    CHART_KEYWORDS = ["图表", "趋势", "曲线", "柱状", "饼图", "折线"]
    TABLE_KEYWORDS = ["表格", "数据", "列", "行", "对比"]
    IMAGE_KEYWORDS = ["图片", "图像", "图示", "示意图", "截图"]

    def analyze(self, query: str) -> dict:
        return {
            "needs_chart": any(kw in query for kw in self.CHART_KEYWORDS),
            "needs_table": any(kw in query for kw in self.TABLE_KEYWORDS),
            "needs_image": any(kw in query for kw in self.IMAGE_KEYWORDS),
        }
```
